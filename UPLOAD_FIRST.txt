#!/usr/bin/env python3
"""Create deterministic runtime and source ZIPs for esqc_gps."""
from __future__ import annotations

import argparse
import hashlib
import os
import shutil
import subprocess
import sys
import tempfile
import zipfile
from pathlib import Path, PurePosixPath

ROOT = Path(__file__).resolve().parents[1]
VERSION = "1.2.0"
ZIP_TIME = (1980, 1, 1, 0, 0, 0)
RUNTIME_NAMES = [
    "esqc_gps.ado",
    "esqc_gps.sthlp",
    "esqc_gps.jar",
    "esqc_gps_log.ado",
    "esqc_gps_log.sthlp",
    "esqc_gps.pkg",
    "stata.toc",
    "README.md",
    "README_1.2.0_HTML_MAP.md",
    "SECURITY.md",
    "requirements.txt",
    "examples/standalone_gps.do",
    "MANIFEST.sha256",
]
EXCLUDED_PARTS = {".git", "__pycache__", "build", ".pytest_cache"}
EXCLUDED_SUFFIXES = {".pyc", ".pyo"}


def sha256(path: Path) -> str:
    digest = hashlib.sha256()
    with path.open("rb") as handle:
        for chunk in iter(lambda: handle.read(1024 * 1024), b""):
            digest.update(chunk)
    return digest.hexdigest()


def release_files() -> list[Path]:
    files: list[Path] = []
    for path in ROOT.rglob("*"):
        if not path.is_file():
            continue
        rel = path.relative_to(ROOT)
        if any(part in EXCLUDED_PARTS for part in rel.parts):
            continue
        if path.suffix.lower() in EXCLUDED_SUFFIXES:
            continue
        if rel.as_posix() == "MANIFEST.sha256":
            continue
        files.append(path)
    return sorted(files, key=lambda p: p.relative_to(ROOT).as_posix())


def write_manifest() -> None:
    lines = ["# SHA-256 manifest for esqc_gps 1.2.0; this file excludes itself."]
    for path in release_files():
        rel = path.relative_to(ROOT).as_posix()
        lines.append(f"{sha256(path)}  {rel}")
    (ROOT / "MANIFEST.sha256").write_text("\n".join(lines) + "\n", encoding="utf-8")


def info(name: str) -> zipfile.ZipInfo:
    pure = PurePosixPath(name)
    if pure.is_absolute() or ".." in pure.parts or "\\" in name:
        raise ValueError(f"unsafe ZIP member: {name}")
    result = zipfile.ZipInfo(name, ZIP_TIME)
    result.compress_type = zipfile.ZIP_DEFLATED
    result.create_system = 3
    result.external_attr = (0o100644 & 0xFFFF) << 16
    result.flag_bits |= 0x800
    return result


def write_zip(output: Path, members: list[tuple[str, Path]]) -> None:
    output.parent.mkdir(parents=True, exist_ok=True)
    fd, temp_name = tempfile.mkstemp(prefix=output.name + ".", suffix=".tmp", dir=str(output.parent))
    os.close(fd)
    temp = Path(temp_name)
    try:
        with zipfile.ZipFile(temp, "w", compression=zipfile.ZIP_DEFLATED, compresslevel=9) as archive:
            for name, path in sorted(members):
                archive.writestr(info(name), path.read_bytes(), compress_type=zipfile.ZIP_DEFLATED, compresslevel=9)
        os.replace(temp, output)
    except Exception:
        temp.unlink(missing_ok=True)
        raise


def main() -> int:
    parser = argparse.ArgumentParser()
    parser.add_argument("--outdir", type=Path, default=ROOT.parent)
    parser.add_argument("--skip-validation", action="store_true")
    args = parser.parse_args()
    outdir = args.outdir.resolve()

    subprocess.run([sys.executable, str(ROOT / "tools" / "build_jar.py")], cwd=ROOT, check=True)
    write_manifest()

    runtime_output = outdir / f"esqc_gps_stata_runtime_{VERSION}.zip"
    source_output = outdir / f"esqc_gps_luna_source_{VERSION}.zip"
    runtime_root = f"esqc_gps_stata_runtime_{VERSION}"
    source_root = f"esqc_gps_luna_source_{VERSION}"

    runtime_members = []
    for name in RUNTIME_NAMES:
        path = ROOT / name
        if not path.is_file():
            raise FileNotFoundError(path)
        runtime_members.append((f"{runtime_root}/{name}", path))
    write_zip(runtime_output, runtime_members)

    source_members = [
        (f"{source_root}/{path.relative_to(ROOT).as_posix()}", path)
        for path in release_files() + [ROOT / "MANIFEST.sha256"]
    ]
    write_zip(source_output, source_members)

    if not args.skip_validation:
        subprocess.run(
            [
                sys.executable,
                str(ROOT / "tools" / "validate_release.py"),
                "--zip", str(runtime_output),
                "--zip", str(source_output),
            ],
            cwd=ROOT,
            check=True,
        )

    print(f"runtime: {runtime_output} ({runtime_output.stat().st_size} bytes, sha256={sha256(runtime_output)})")
    print(f"source:  {source_output} ({source_output.stat().st_size} bytes, sha256={sha256(source_output)})")
    return 0


if __name__ == "__main__":
    raise SystemExit(main())

def clockwise_square(minx: float, miny: float, maxx: float, maxy: float) -> list[list[float]]:
    return [
        [minx, miny],
        [minx, maxy],
        [maxx, maxy],
        [maxx, miny],
        [minx, miny],
    ]


def write_layer(directory: Path, *, epsg: int = 4326, write_prj: bool = True) -> Path:
    directory.mkdir(parents=True, exist_ok=True)
    base = directory / "test_adm2"
    writer = shapefile.Writer(str(base), shapeType=shapefile.POLYGON, encoding="utf-8")
    writer.field("NAM_0", "C", size=80)
    writer.field("NAM_1", "C", size=80)
    writer.field("NAM_2", "C", size=80)

    polygons = [
        (clockwise_square(10.0, 10.0, 11.0, 11.0), "North", "Alpha"),
        (clockwise_square(11.0, 10.0, 12.0, 11.0), "South", "Beta"),
    ]
    if epsg != 4326:
        transformer = Transformer.from_crs(4326, epsg, always_xy=True)
        projected = []
        for ring, admin1, admin2 in polygons:
            projected.append(([[*transformer.transform(x, y)] for x, y in ring], admin1, admin2))
        polygons = projected

    for ring, admin1, admin2 in polygons:
        writer.poly([ring])
        writer.record("Testland", admin1, admin2)
    writer.close()
    (base.with_suffix(".cpg")).write_text("UTF-8\n", encoding="ascii")
    if write_prj:
        base.with_suffix(".prj").write_text(CRS.from_epsg(epsg).to_wkt(), encoding="utf-8")
    return base.with_suffix(".shp")


def write_input(path: Path, rows: list[dict[str, Any]]) -> None:
    with path.open("w", encoding="utf-8") as handle:
        for row in rows:
            handle.write(json.dumps(row, ensure_ascii=False) + "\n")

def run_worker(
    root: Path,
    shapefile_path: Path,
    rows: list[dict[str, Any]],
    *,
    extra: dict[str, Any] | None = None,
    env: dict[str, str] | None = None,
) -> tuple[subprocess.CompletedProcess[str], Path, Path, Path, Path]:
    root.mkdir(parents=True, exist_ok=True)
    input_path = root / "input.jsonl"
    output_path = root / "result.csv"
    marker_path = root / "success.marker"
    cache_path = root / "cache.json"
    config_path = root / "config.json"
    write_input(input_path, rows)
    config: dict[str, Any] = {
        "INPUT_JSONL": str(input_path),
        "OUTPUT_CSV": str(output_path),
        "SUCCESS_MARKER": str(marker_path),
        "SHP_PATH": str(shapefile_path),
        "COUNTRY_NAME": "Testland",
        "COUNTRY_FIELD": "NAM_0",
        "ADMIN1_FIELD": "NAM_1",
        "ADMIN2_FIELD": "NAM_2",
        "ASSUME_WGS84": False,
        "CACHE_FILE": str(cache_path),
        "AI_VERIFY": False,
        "AI_MODEL": "gpt-5.6-luna",
        "VISION_ENABLED": False,
        "WEB_SEARCH": False,
        "BASEMAP": "osm",
        "REASONING_EFFORT": "medium",
        "AI_BATCH_SIZE": 10,
        "VISION_BATCH_SIZE": 3,
        "HTTP_TIMEOUT_SECS": 30,
        "SNAP_TOLERANCE_KM": 0,
    }
    if extra:
        config.update(extra)
    config_path.write_text(json.dumps(config), encoding="utf-8")
    process_env = os.environ.copy()
    process_env.pop("ESQC_AI_KEY", None)
    process_env.pop("OPENAI_API_KEY", None)
    process_env.setdefault("ESQC_GPS_LOG_DIR", str(root / "full_logs"))
    if env:
        process_env.update(env)
    completed = subprocess.run(
        [sys.executable, str(WORKER), str(config_path)],
        text=True,
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        env=process_env,
        timeout=90,
        check=False,
    )
    return completed, output_path, marker_path, cache_path, config_path


def read_rows(path: Path) -> list[dict[str, str]]:
    with path.open("r", encoding="utf-8", newline="") as handle:
        reader = csv.DictReader(handle)
        if reader.fieldnames != EXPECTED_HEADER:
            raise AssertionError(f"unexpected header: {reader.fieldnames}")
        return list(reader)


class WorkerTests(unittest.TestCase):
    def test_deterministic_contract_statuses_cache_and_zip(self) -> None:
        with tempfile.TemporaryDirectory() as tmp:
            root = Path(tmp)
            shp = write_layer(root / "shape")
            completed, output, marker, cache, _ = run_worker(root / "first", shp, default_rows())
            self.assertEqual(completed.returncode, 0, completed.stderr + completed.stdout)
            self.assertEqual(
                marker.read_text(encoding="utf-8").strip(),
                "ESQC_GPS_OK\tversion=1.2.0\tschema=sfi11\trows=7",
            )
            audit_dir = root / "first" / "full_logs"
            audit_files = list(audit_dir.glob("esqc_gps_*.jsonl"))
            self.assertEqual(len(audit_files), 1)
            audit_text = audit_files[0].read_text(encoding="utf-8")
            self.assertIn('"event":"run_started"', audit_text)
            self.assertIn('"log_policy":"always_full"', audit_text)
            self.assertIn('"openai_store_policy":"forced_true"', audit_text)
            self.assertIn('"event":"run_finished"', audit_text)
            rows = read_rows(output)
            self.assertEqual([int(row["row_id"]) for row in rows], list(range(1, 8)))
            self.assertEqual([row["tid"] for row in rows][0::6], ["dup", "dup"])
            self.assertEqual(
                [row["gps_status"] for row in rows],
                ["ok", "mismatch", "reported_not_found", "outside_country", "missing", "no_reported_location", "ok"],
            )
            self.assertEqual([row["gps_flag"] for row in rows], ["0", "1", "1", "1", "1", "1", "0"])
            self.assertEqual(rows[0]["predicted_admin1"], "North")
            self.assertEqual(rows[0]["predicted_admin2"], "Alpha")
            self.assertEqual(rows[1]["a3a_status"], "mismatch")
            self.assertEqual(rows[2]["a3x_status"], "not_found")
            cache_payload = json.loads(cache.read_text(encoding="utf-8"))
            self.assertEqual(cache_payload["version"], "1.2.0")
            self.assertEqual(len(cache_payload["entries"]), 7)
            self.assertEqual(
                rows[0]["detail"],
                "OK: GPS in Alpha, North; a3a 'North' includes North; a3x 'Alpha' contains GPS.",
            )
            self.assertNotIn("AI alignment", rows[0]["detail"])

            # The same layer inside a ZIP must behave identically; signature and
            # row cache keys do not depend on the temporary extraction path.
            archive = root / "adm2.zip"
            with zipfile.ZipFile(archive, "w", zipfile.ZIP_DEFLATED) as zf:
                for sidecar in sorted((root / "shape").glob("test_adm2.*")):
                    zf.write(sidecar, arcname=f"nested/{sidecar.name}")
            second_root = root / "second"
            second_root.mkdir()
            completed2, output2, marker2, _, _ = run_worker(second_root, archive, default_rows())
            self.assertEqual(completed2.returncode, 0, completed2.stderr + completed2.stdout)
            self.assertTrue(marker2.exists())
            self.assertEqual(
                [(r["row_id"], r["gps_status"]) for r in read_rows(output2)],
                [(r["row_id"], r["gps_status"]) for r in rows],
            )

    def test_missing_crs_fails_closed_unless_assumed(self) -> None:
        with tempfile.TemporaryDirectory() as tmp:
            root = Path(tmp)
            shp = write_layer(root / "shape", write_prj=False)
            rows = [{"row_id": 1, "tid": "p", "lat": 10.5, "lon": 10.5, "a3a": "North", "a3x": "Alpha"}]

            self.assertEqual(passed.returncode, 0, passed.stderr + passed.stdout)
            self.assertTrue(marker2.exists())
            self.assertEqual(read_rows(output)[0]["gps_status"], "ok")

    def test_zip_path_traversal_is_rejected(self) -> None:
        with tempfile.TemporaryDirectory() as tmp:
            root = Path(tmp)
            archive = root / "bad.zip"
            with zipfile.ZipFile(archive, "w") as zf:

    def test_ai_request_uses_luna_responses_structured_output_and_current_cache_fields(self) -> None:
        requests: list[dict[str, Any]] = []

        class Handler(BaseHTTPRequestHandler):
            protocol_version = "HTTP/1.1"

            def do_POST(self) -> None:  # noqa: N802 - stdlib hook
                length = int(self.headers.get("Content-Length", "0"))
                body = json.loads(self.rfile.read(length).decode("utf-8"))
                requests.append({"path": self.path, "headers": dict(self.headers), "body": body})
                payload = {
                    "id": "resp_test",
                    "object": "response",
                    "status": "completed",
                    "output": [
                        {
                            "type": "message",
                            "role": "assistant",
                            "content": [
                                {
                                    "type": "output_text",
                                    "text": json.dumps(
                                        {
                                            "results": [
                                                {
                                                    "case_id": "c1",
                                                    "a3a_status": "match",
                                                    "a3a_confidence": "medium",
                                                    "a3a_reason": "Common Admin1 abbreviation.",
                                                    "a3x_status": "match",
                                                    "a3x_confidence": "high",
                                                    "a3x_reason": "Locked deterministic locality match.",
                                                }
                                            ]
                                        }
                                    ),
                                }
                            ],
                        }
                    ],
                    "usage": {
                        "input_tokens": 100,
                        "input_tokens_details": {"cached_tokens": 0, "cache_write_tokens": 100},
                        "output_tokens": 20,
                        "output_tokens_details": {"reasoning_tokens": 5},
                        "total_tokens": 120,
                    },
                }
                encoded = json.dumps(payload).encode("utf-8")
                self.send_response(200)
                self.send_header("Content-Type", "application/json")
                self.send_header("Content-Length", str(len(encoded)))
                self.end_headers()
                self.wfile.write(encoded)

            def log_message(self, fmt: str, *args: Any) -> None:
                return

        server = ThreadingHTTPServer(("127.0.0.1", 0), Handler)
        thread = threading.Thread(target=server.serve_forever, daemon=True)
        thread.start()
        try:
            with tempfile.TemporaryDirectory() as tmp:
                root = Path(tmp)
                shp = write_layer(root / "shape")
                # Deterministically a3a is unresolved, so it enters the text AI pass.
                rows = [{"row_id": 1, "tid": "ai", "lat": 10.5, "lon": 10.5, "a3a": "N. region", "a3x": "Alpha"}]
                completed, output, marker, _, _ = run_worker(
                    root / "run",
                    shp,
                    rows,
                    extra={
                        "AI_VERIFY": True,
                        "AI_MODEL": "gpt-5.6-luna",
                        "VISION_ENABLED": False,
                        "WEB_SEARCH": True,
                        "REASONING_EFFORT": "medium",
                    },
                    env={
                        "ESQC_AI_KEY": "sk-test-secret",
                        "OPENAI_BASE_URL": f"http://127.0.0.1:{server.server_port}/v1",
                    },
                )
                self.assertEqual(completed.returncode, 0, completed.stderr + completed.stdout)
                self.assertTrue(marker.exists())
                self.assertEqual(read_rows(output)[0]["gps_status"], "ok")

            self.assertEqual(len(requests), 1)
            request = requests[0]
            self.assertEqual(request["path"], "/v1/responses")
            self.assertEqual({k.lower(): v for k, v in request["headers"].items()}.get("authorization"), "Bearer sk-test-secret")
            body = request["body"]
            self.assertEqual(body["model"], "gpt-5.6-luna")
            self.assertIs(body["store"], True)
            self.assertEqual(body["reasoning"], {"effort": "medium"})
            self.assertEqual(body["prompt_cache_options"], {"mode": "implicit", "ttl": "30m"})
            self.assertNotIn("prompt_cache_retention", body)
            self.assertTrue(body["prompt_cache_key"].startswith("esqc:gps:gps_align:"))
            self.assertEqual(body["tools"], [{"type": "web_search", "search_context_size": "low"}])
            self.assertEqual(body["tool_choice"], "auto")
            self.assertIs(body["parallel_tool_calls"], False)
            fmt = body["text"]["format"]
            self.assertEqual(fmt["type"], "json_schema")
            self.assertEqual(fmt["name"], "esqc_gps_alignment")
            self.assertIs(fmt["strict"], True)
            self.assertEqual(fmt["schema"]["additionalProperties"], False)
            item_schema = fmt["schema"]["properties"]["results"]["items"]
            self.assertIn("a3x_confidence", item_schema["required"])
            self.assertIn("a3x_reason", item_schema["required"])
        finally:
            server.shutdown()
            server.server_close()
            thread.join(timeout=5)
